\begin{exer}{(Problem 3)}
  \begin{align*}
    \int_a^b \abs{z'(t)} dt
      &= \int_c^d \abs{z'(t(s))}t'(s) ds \\
      &= \int_c^d \abs{z'(t(s))t'(s)} ds \\
      &= \int_c^d \abs{\tilde{z}'(s)} ds
  \end{align*}
  
  where $\tilde{z}(s): [c, d] \rightarrow \mathbb{C}$ is a reparametrization of $z(t): [a, b] \rightarrow \mathbb{C}$.
\end{exer}
