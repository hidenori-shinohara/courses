\title{Antiderivative problems}
\author{Hidenori Shinohara}
\maketitle

\begin{exer}{(Problem A)}
  Suppose $f'' = 12x^2 + 6x - 4$ and $f(0) = 4$ and $f(1) = 1$.
  Find $f$.
\end{exer}

\begin{exer}{(Problem B)}
  Find the most general antiderivative of $4\sin x + \frac{2x^5 + 1}{x}$.
\end{exer}

\begin{exer}{(Problem C)}
  Find $f(x)$ such that $f'(x) = \cos x$.
\end{exer}

\begin{exer}{(Problem D)}
  Is $x^2 + 3x - 2$ an antiderivative of $2x - 3$?
\end{exer}
