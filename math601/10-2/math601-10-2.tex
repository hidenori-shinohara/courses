\section{The Quadratic Equation $x^2 - 2y^2 = n$}
\begin{exer}{(Problem 15)}
  Find a solution to $x^2 - 2y^2 = 7$.
\end{exer}

\begin{proof}
  $3^2 - 2 \cdot 1^2 = 9 - 2 = 7$.
  Thus $(x, y) = (3, 1)$ is a solution to $x^2 - 2y^2 = 7$.
\end{proof}
